import context from "./context"
import { settings } from "./settings"
import placements from "./placements/index"
import internal from "./api/internal"

/**
 * @constructor
 */
function Api() {
  this.internal = internal
  this.placements = placements

  /**
   * @deprecated since this was a quick hack for usage in Codepen.IO
   */
  this.setResponseMode = () => {
    // No Op
  }

  /**
   * @return {Session}
   */
  this.createSession = () => {
    return internal.createSession()
  }

  /**
   * @return {Session}
   */
  this.defaultSession = () => {
    return internal.defaultSession()
  }

  this.createRecommendationRequest = flags => {
    return internal.createRecommendationRequest(flags)
  }

  this.setAutoLoad = flag => {
    internal.setAutoLoad(flag)
  }

  this.isAutoLoad = () => {
    return internal.isAutoLoad()
  }

  this.setRecommendationsEnabled = flag => {
    internal.setRecommendationsEnabled(flag)
  }

  /**
   * @param phase
   * @param cb
   */
  this.listen = (phase, cb) => {
    internal.listen(phase, cb)
  }

  /**
   * @return {Promise}
   */
  this.loadRecommendations = () => {
    return internal.loadRecommendations()
  }

  this.pageTagging = () => {
    return internal.extractTagging()
  }

  this.loadCartPopupRecommendations = (products, cart, alwaysShow) => {
    return internal.loadCartPopupRecommendations(products, cart, alwaysShow)
  }

  // renamed function reportAddToCart to recommendedProductAddedToCart (still keeping old for backwards compatibility)
  // The name of this function has caused lot's of confusion because people assume this is used to track cart status
  // cart status is tracked through the cart tagging, this is just used to mark attribution when a dynamically rendered
  // nosto element includes add-to-cart button and customer uses that to add the recommended product to their cart
  this.reportAddToCart = (cartItemId, nostoElementId) => {
    return internal.recommendedProductAddedToCart(cartItemId, nostoElementId)
  }

  /**
   * @param {String} productId
   * @param {String} nostoElementId
   * @return {Promise<Object>}
   */
  this.recommendedProductAddedToCart = (productId, nostoElementId) => {
    return internal.recommendedProductAddedToCart(productId, nostoElementId)
  }

  this.experiments = experiments => {
    return internal.setExperiments(experiments)
  }

  this.customer = customer => {
    return internal.setCustomer(customer)
  }

  this.popupCampaigns = () => {
    return internal.popupCampaigns()
  }

  this.reloadOverlay = () => {
    internal.reloadOverlay()
  }

  this.openPopup = (popupId, options) => {
    internal.openPopup(popupId, options)
  }

  this.enablePopup = popupId => {
    internal.enablePopup(popupId)
  }

  this.disablePopup = popupId => {
    internal.disablePopup(popupId)
  }

  /**
   * API method to resend the cart content to Nosto. This is used in situations
   * when the cart tagging is loaded after the client script initialization.
   * Look at an Magento 2 merchant for a example on how this is used.
   *
   * @param cart content of the cart
   */
  this.resendCartContent = cart => {
    return internal.resendCartContent(cart)
  }

  /**
   * API method to resend the cart tagging to Nosto. This is used in situations
   * when the cart tagging is loaded after the client script initialization.
   * Look at an Magento 2 merchant for a example on how this is used.
   */
  this.resendCartTagging = () => {
    return internal.resendCartTagging()
  }

  /**
   * API method to resend the customer tagging to Nosto. This is used in situations
   * when the customer tagging is loaded after the client script initialization.
   * Look at an Magento 2 merchant for a example on how this is used.
   */
  this.resendCustomerTagging = () => {
    return internal.resendCustomerTagging()
  }

  this.sendTagging = (ref, opts) => {
    return internal.sendTagging(ref, opts)
  }
  /**
   * @param {String} segment
   * @return {Promise<Object>}
   */
  this.addSegmentCodeToVisit = segment => {
    return internal.addSegment(segment)
  }

  this.removeCampaigns = (unFilledElements, dynamicPlacements) => {
    internal.removeCampaigns(unFilledElements, dynamicPlacements)
  }

  this.showPlacementPreviews = (placement, content) => {
    internal.showPlacementPreviews(placement, content)
  }

  this.install = callbackFn => {
    // noinspection JSUnusedGlobalSymbols
    callbackFn({
      context,
      settings,
      overlay: internal.getOverlay()
    })
  }
}

export default new Api()
